\examsection{一、选择题：本题共12小题，每小题5分，共60分. 在每小题给出的四个选项中，只有一项是符合题目要求的.}
\begin{examquestions}

\item 已知全集 $U=\{1,2,3,4,5\}$，集合 $M=\{1,2\}$，$N=\{3,4\}$，则 $\complement_U(M\cup N)=$
\fourchoices{$\{5\}$}{$\{1,2\}$}{$\{3,4\}$}{$\{1,2,3,4\}$}

\item 设 $\mathrm{i}z=4+3\mathrm{i}$，则 $z=$
\fourchoices{$-3-4\mathrm{i}$}{$-3+4\mathrm{i}$}{$3-4\mathrm{i}$}{$3+4\mathrm{i}$}

\item 已知命题 $p:\exists x\in\mathbb R,\ \sin x<1$；命题 $q:\forall x\in\mathbb R,\ e^{|x|}\geq1$，则下列命题中为真命题的是
\fourchoices{$p\wedge q$}{$\neg p\wedge q$}{$p\wedge\neg q$}{$\neg(p\vee q)$}

\item 函数 $f(x)=\sin\dfrac{x}{3}+\cos\dfrac{x}{3}$ 的最小正周期和最大值分别是
\fourchoices{$3\pi$ 和 $\sqrt2$}{$3\pi$ 和 $2$}{$6\pi$ 和 $\sqrt2$}{$6\pi$ 和 $2$}

\item 若 $x,y$ 满足约束条件\linsys{x+y\geq4\\x-y\leq2\\y\leq3}，则 $z=3x+y$ 的最小值为
\fourchoices{$18$}{$10$}{$6$}{$4$}

\item $\cos^2\dfrac{\pi}{12}-\cos^2\dfrac{5\pi}{12}=$
\fourchoices
{\tallchoice{$\dfrac12$}}
{\tallchoice{$\dfrac{\sqrt3}{3}$}}
{\tallchoice{$\dfrac{\sqrt2}{2}$}}
{\tallchoice{$\dfrac{\sqrt3}{2}$}}

\item 在区间 $\left(0,\dfrac12\right)$ 随机取 $1$ 个数，则取到的数小于 $\dfrac13$ 的概率为
\fourchoices{\tallchoice{$\dfrac34$}}{\tallchoice{$\dfrac23$}}{\tallchoice{$\dfrac13$}}{\tallchoice{$\dfrac16$}}

\item 下列函数中最小值为 $4$ 的是
\twochoices
{$y=x^2+2x+4$}
{\tallchoice{$y=|\sin x|+\dfrac4{|\sin x|}$}}
{$y=2^x+2^{2-x}$}
{\tallchoice{$y=\ln x+\dfrac4{\ln x}$}}

\item 设函数 $f(x)=\dfrac{1-x}{1+x}$，则下列函数中为奇函数的是
\fourchoices{$f(x-1)-1$}{$f(x-1)+1$}{$f(x+1)-1$}{$f(x+1)+1$}

\item 在正方体 $ABCD-A_1B_1C_1D_1$ 中，$P$ 为 $B_1D_1$ 的中点，则直线 $PB$ 与 $AD_1$ 所成的角为
\fourchoices
{\tallchoice{$\dfrac{\pi}{2}$}}
{\tallchoice{$\dfrac{\pi}{3}$}}
{\tallchoice{$\dfrac{\pi}{4}$}}
{\tallchoice{$\dfrac{\pi}{6}$}}

\item 设 $B$ 是椭圆 $C:\dfrac{x^2}{5}+y^2=1$ 的上顶点，点 $P$ 在 $C$ 上，则 $|PB|$ 的最大值为
\fourchoices{\tallchoice{$\dfrac52$}}{$\sqrt6$}{$\sqrt5$}{$2$}

\item 设 $a\ne0$，若 $x=a$ 为函数 $f(x)=a(x-a)^2(x-b)$ 的极大值点，则
\fourchoices{$a<b$}{$a>b$}{$ab<a^2$}{$ab>a^2$}

\end{examquestions}
