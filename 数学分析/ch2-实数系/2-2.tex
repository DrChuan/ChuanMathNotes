\section{实数系公理}

实数系 $(\mathbb{R}, +, \cdot, \le)$ 满足如下公理. 设 $x,y,z \in \mathbb{R}$. 

\textbf{(F) 域公理}
\begin{itemize}
    \item 加法：\((A1)\) 加法结合律；\((A2)\) 加法交换律；\((A3)\) 存在加法单位元（零元）；\((A4)\) 存在负元. 
    \item 乘法：\((M1)\) 乘法结合律；\((M2)\) 乘法交换律；\((M3)\) 存在乘法单位元；\((M4)\) 非零元存在逆元. 
    \item \((AM)\) 乘法分配律. 
\end{itemize}

\textbf{(O) 序公理}
\begin{itemize}
    \item \((O1)\) 反对称性；
    \item \((O2)\) 传递性；
    \item \((O3)\) 分离性；
    \item \((AO)\) 加法保序性；
    \item \((MO)\) 乘法保序性. 
\end{itemize}

\textbf{(C) 连续公理}
\begin{itemize}
    \item \((C1)\) 阿基米德公理；
    \item \((C2)\) 完备公理. 
\end{itemize}
